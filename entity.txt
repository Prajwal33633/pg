package com.infy.tracking_service.entity;

import java.time.LocalDate;
import java.time.LocalDateTime;

import jakarta.persistence.Column;
import jakarta.persistence.Entity;
import jakarta.persistence.Id;
import jakarta.persistence.JoinColumn;
import jakarta.persistence.ManyToOne;
import jakarta.persistence.Table;
import lombok.AllArgsConstructor;
import lombok.Data;
import lombok.NoArgsConstructor;

@Entity
@Data
@NoArgsConstructor
@AllArgsConstructor
@Table(name = "package")
public class Package_E {
    @Id
    @Column(length = 255, name = "tracking_number")
    private String trackingNumber;

    @Column(length = 255, name = "current_status")
    private String currentStatus;

    @Column(length = 255, name = "last_known_location")
    private String lastKnownLocation;

    @Column(name = "timestamp")
    private LocalDateTime timestamp;

    @Column(name = "estimated_delivery_date")
    private LocalDate estimatedDeliveryDate;

    @Column(length = 255, name = "package_type")
    private String packageType;

    @Column(length = 255, name = "destination")
    private String destination;

    @Column(length = 255, name = "priority")
    private String priority;

    @Column(name = "weight")
    private Double weight;

    @Column(length = 255, name = "dimensions")
    private String dimensions;

    @Column(length = 255, name = "service_level")
    private String serviceLevel;

    @Column(length = 255, name = "origin")
    private String origin;

    @Column(name = "shipping_cost")
    private Double shippingCost;

    @Column(length = 255, name = "return_tracking_number")
    private String returnTrackingNumber;

    @Column(length = 255, name = "original_tracking_number")
    private String originalTrackingNumber;

    @Column(length = 255, name = "damage_details")
    private String damageDetails;

    @Column(name = "is_oversized")
    private Boolean isOversized;

    @Column(name = "is_suspicious")
    private Boolean isSuspicious;

    @ManyToOne
    @JoinColumn(name = "customer_id")
    private Customer customer;

    @ManyToOne
    @JoinColumn(name = "distribution_center_id")
    private DistributionCenter distributionCenter;
}
